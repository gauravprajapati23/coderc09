{

    int cost[MAX][MAX];

    int u, v, min_distance, distance[MAX], from[MAX];

    int visited[MAX], no_of_edges, i, min_cost, j;

    for (i = 0; i < n; i++)

        for (j = 0; j < n; j++)

        {

            if (G[i][j] == 0)

                cost[i][j] = infinity;

            else

                cost[i][j] = G[i][j];

            spanning[i][j] = 0;

        }

    distance[0] = 0;

    visited[0] = 1;

    for (i = 1; i < n; i++)

    {

        distance[i] = cost[0][i];

        from[i] = 0;

        visited[i] = 0;

    }

    min_cost = 0;

    no_of_edges = n - 1;

    while (no_of_edges > 0)

    {

        min_distance = infinity;

        for (i = 1; i < n; i++)

            if (visited[i] == 0 && distance[i] < min_distance)

            {

                v = i;

                min_distance = distance[i];

            }

        u = from[v];

        spanning[u][v] = distance[v];

        spanning[v][u] = distance[v];

        no_of_edges--;

        visited[v] = 1;

        for (i = 1; i < n; i++)

            if (visited[1] == 0 && cost[i][v] < distance[i])

            {

                distance[i] = cost[i][v];

                from[i] = v;

            }

        min_cost = min_cost + cost[u][v];

    }

    return (min_cost);

}

{

    int tmp = *a;

    *a = *b;

void heapify(int arr[], int n, int i)

{

    int max = i;

    int leftChild = 2 * i;

    int rightChild = 2 * i + 1;

    if (leftChild < n && arr[leftChild] > arr[max])

        max = leftChild;

    if (rightChild < n && arr[rightChild] > arr[max])

        max = rightChild;

    if (max != i)

    {

        swap(&arr[i], &arr[max]);

        heapify(arr, n, max);

    }

}

{

    for (int i = n / 2 - 1; i >= 0; i--)

        swap(&arr[0], &arr[i]);

        heapify(arr, i, 0);

}

{

        max = rightChild;

}
